%============================
\section{Giới thiệu project}
%============================

\begin{frame}{Chúng tôi đang làm gì?}
  \textbf{Sản phẩm:} Xây dựng \textbf{Hệ thống kiểm soát ra vào thông minh} (Smart Access Control) -- một cánh cửa an ninh tự động \textbf{nhận diện khuôn mặt} để cho phép hoặc từ chối người ra vào.

  \vspace{0.8em}
  \begin{itemize}
    \item Đây là một hệ thống \textbf{AIoT}: kết hợp Trí tuệ nhân tạo (AI) với Internet vạn vật (IoT).
    \item Phần cứng lõi: \textbf{ESP32-S3-CAM} -- vừa chụp ảnh, vừa chạy mô hình AI ngay trên chip (\textit{Edge AI}).
    \item Cơ cấu chấp hành: \textbf{Servo} chỉnh góc để đóng/mở cánh cửa (mô hình bằng bìa).
    \item Giám sát từ xa: gửi nhật ký và cảnh báo qua \textbf{Wi-Fi} tới Dashboard / Telegram.
  \end{itemize}
\end{frame}

\begin{frame}{Dùng cho việc gì?}
  \textbf{Đối tượng và bối cảnh sử dụng:} các \textbf{không gian giới hạn}, quy mô nhỏ (khoảng 15--20 người) như Phòng Lab, văn phòng nhỏ, phòng server, phòng nghiên cứu.

  \vspace{0.8em}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \textbf{Vấn đề hiện tại}
      \begin{itemize}
        \item Chìa khóa / thẻ từ dễ mất, dễ sao chép
        \item Khó biết ai đã ra vào, lúc nào
        \item Khó cảnh báo khi có người lạ
      \end{itemize}

    \column{0.5\textwidth}
      \textbf{Giải pháp đề xuất}
      \begin{itemize}
        \item Khuôn mặt là ``chìa khóa'' -- không thể quên
        \item Tự động ghi nhật ký ra/vào
        \item Cảnh báo kèm ảnh khi phát hiện người lạ
      \end{itemize}
  \end{columns}
\end{frame}

\section{Mục đích cần đạt được}
%============================

\begin{frame}{Mục đích}
  Xây dựng một hệ thống kiểm soát ra vào \textbf{tự động, an toàn và phản hồi nhanh}, hoạt động theo cơ chế kết hợp chặt chẽ giữa \textbf{Edge AI} và \textbf{IoT}.

  \vspace{0.8em}
  \begin{itemize}
    \item Toàn bộ việc chụp ảnh, phát hiện và nhận diện khuôn mặt được xử lý \textbf{trực tiếp trên chip} (Local), không cần truyền video stream lên Cloud.
    \item Điều này giúp \textbf{tối ưu bảo mật} hình ảnh cá nhân và \textbf{giảm tối đa độ trễ} phản hồi.
    \item Khi xác thực thành công, hệ thống \textbf{tự động} đóng/mở khóa và \textbf{ghi nhận} sự kiện về trung tâm quản lý.
  \end{itemize}
\end{frame}

\section{Các mục tiêu của project}
%============================

\begin{frame}{Mục tiêu theo chuẩn SMART}
  Các mục tiêu được lượng hóa cụ thể theo nguyên tắc \textbf{S}pecific, \textbf{M}easurable, \textbf{A}chievable, \textbf{R}elevant, \textbf{T}ime-bound.

  \vspace{0.8em}
  \renewcommand{\arraystretch}{1.35}
  \begin{tabular}{p{0.6cm}p{6.0cm}p{5.5cm}}
    \toprule
    \textbf{MT} & \textbf{Nội dung} & \textbf{Chỉ tiêu chính} \\
    \midrule
    1 & Phát hiện khuôn mặt \newline (Face Detection) & Độ chính xác $>95\%$ ở 0.5--1.5\,m \\
    2 & Nhận dạng khuôn mặt \newline (Face Recognition) & Định danh $>92\%$ trên 20 người, FAR $<2\%$, phản hồi $<1.5$\,s \\
    3 & Kết nối và điều khiển IoT & Tỷ lệ truyền nhận thành công $99\%$, độ trễ $<2$\,s \\
    \bottomrule
  \end{tabular}
\end{frame}

\begin{frame}{Mục tiêu 1 -- Phát hiện khuôn mặt}
  \textbf{Face Detection}

  \vspace{0.4em}
  Triển khai thành công mô hình quét ảnh, phát hiện sự xuất hiện của khuôn mặt người trong khung hình.

  \vspace{0.8em}
  \begin{itemize}
    \item Tỷ lệ phát hiện chính xác \textbf{$>95\%$} (ở khoảng cách \textbf{0.5m -- 1.5m}).
    \item Đảm bảo \textbf{không bỏ sót} đối tượng bước đến cửa.
  \end{itemize}
\end{frame}

\begin{frame}{Mục tiêu 2 -- Nhận dạng khuôn mặt}
  \textbf{Face Recognition}

  \vspace{0.4em}
  Triển khai mô hình đối khớp đặc trưng để định danh chính xác người dùng trong database.

  \vspace{0.8em}
  \begin{itemize}
    \item Tỷ lệ định danh chính xác \textbf{$>92\%$} trên database \textbf{20 người}.
    \item Tỷ lệ nhận nhầm người (False Acceptance) \textbf{$<2\%$}.
    \item Thời gian phản hồi \textbf{$<1.5$ giây}.
  \end{itemize}
\end{frame}

\begin{frame}{Mục tiêu 3 -- Kết nối và Điều khiển IoT}
  \textbf{IoT Connectivity \& Control}

  \vspace{0.4em}
  Thiết kế mạch điều khiển Servo và đồng bộ dữ liệu ra vào về Dashboard/Telegram qua Wi-Fi.

  \vspace{0.8em}
  \begin{itemize}
    \item Tỷ lệ truyền nhận tin thành công \textbf{$99\%$}.
    \item Độ trễ tín hiệu \textbf{$<2$ giây}.
  \end{itemize}
\end{frame}

\section{Phương pháp đề xuất}
%============================

\begin{frame}{Lựa chọn Mô hình AI (Edge AI)}
  \textbf{Framework: ESP-WHO} (Espressif). Để tối ưu hiệu năng trên ESP32-S3-CAM, hệ thống dùng cấu trúc mô hình \textbf{hai giai đoạn}.

  \vspace{0.8em}
  \begin{center}
  \begin{tikzpicture}[
    pipestep/.style={rectangle, rounded corners=4pt, draw=blue!70, fill=blue!8,
      text width=3.0cm, align=center, font=\small, minimum height=1.1cm},
    arr/.style={-{Stealth[length=2.5mm]}, thick, blue!70}
  ]
    \node[pipestep, fill=teal!8, draw=teal!70!black] (cam) {Camera\\stream};
    \node[pipestep, right=0.7cm of cam] (detect) {GĐ 1: Face Detection\\{\scriptsize tìm \& cắt vùng mặt}};
    \node[pipestep, right=0.7cm of detect] (rec) {GĐ 2: Face Recognition\\{\scriptsize định danh người dùng}};
    \node[pipestep, right=0.7cm of rec, fill=orange!10, draw=orange!80!black] (out) {PASS / DENY};

    \draw[arr] (cam) -- (detect);
    \draw[arr] (detect) -- node[above, font=\scriptsize]{bounding box} (rec);
    \draw[arr] (rec) -- (out);
  \end{tikzpicture}
  \end{center}
\end{frame}

\begin{frame}{Giai đoạn 1 -- Phát hiện khuôn mặt}
  \textbf{Face Detection} -- Mô hình: Human Face Detector (MNN) hoặc MobileNetV2-SSD (phiên bản rút gọn).

  \vspace{0.8em}
  \textbf{Nhiệm vụ:}
  \begin{itemize}
    \item Xác định \textbf{vị trí} khuôn mặt trong luồng camera theo \textbf{thời gian thực}.
    \item Cắt ra \textbf{vùng khuôn mặt} (Bounding Box) từ khung hình.
    \item Chuẩn hóa vùng mặt để đưa sang giai đoạn nhận dạng.
  \end{itemize}
\end{frame}

\begin{frame}{Giai đoạn 2 -- Nhận dạng khuôn mặt}
  \textbf{Face Recognition} -- Mô hình: ESP-FaceRecognizer (dựa trên kiến trúc dạng \textbf{MobileFaceNet}).

  \vspace{0.8em}
  \textbf{Cơ chế hoạt động:}
  \begin{itemize}
    \item Trích xuất khuôn mặt thành một \textbf{vector đặc trưng} (feature vector).
    \item \textbf{So sánh khoảng cách hình học} của vector với dữ liệu đã lưu sẵn.
    \item Nếu khoảng cách đủ nhỏ (dưới ngưỡng) $\to$ \textbf{định danh} người dùng.
  \end{itemize}

  \vspace{0.6em}
  \textbf{Vì sao chọn cách này?} MobileFaceNet nhẹ, tối ưu cho thiết bị nhúng $\to$ chạy được \textbf{ngay trên ESP32-S3} mà vẫn đạt độ chính xác cao.
\end{frame}

\begin{frame}{Hướng đi của nhóm -- Web Node-RED điều phối}
  Nhóm chọn xây dựng \textbf{Web quản lý bằng Node-RED} để điều phối việc đăng ký nhân viên, sinh vector đặc trưng và đồng bộ xuống thiết bị.

  \vspace{0.6em}
  \begin{center}
  \begin{tikzpicture}[
    pipestep/.style={rectangle, rounded corners=4pt, draw=blue!70, fill=blue!8,
      text width=2.5cm, align=center, font=\scriptsize, minimum height=1.0cm},
    arr/.style={-{Stealth[length=2.5mm]}, thick, blue!70}
  ]
    \node[pipestep, fill=orange!10, draw=orange!80!black] (web) {1. Web Node-RED\\nhân viên upload ảnh};
    \node[pipestep, right=0.5cm of web] (server) {2. Backend\\trích xuất vector};
    \node[pipestep, right=0.5cm of server, fill=violet!10, draw=violet!70!black] (db) {3. DB + Sync\\xuống ESP32 (MQTT/HTTP)};
    \node[pipestep, right=0.5cm of db, fill=teal!8, draw=teal!70!black] (esp) {4. ESP32-S3\\so khớp tại chỗ \& xoay Servo};

    \draw[arr] (web) -- (server);
    \draw[arr] (server) -- (db);
    \draw[arr] (db) -- (esp);
  \end{tikzpicture}
  \end{center}

  \vspace{0.4em}
  \textbf{Điểm mấu chốt:} ảnh gốc chỉ nằm trên Server; ESP32 chỉ lưu \textbf{vector mã hóa} trong Flash/NVS.
\end{frame}

\begin{frame}{Luồng hoạt động chi tiết}
  \begin{enumerate}
    \item Nhân viên truy cập \textbf{Web quản lý (Node-RED)} của công ty, nhập thông tin và upload ảnh đại diện.
    \item \textbf{Server (Backend)} nhận ảnh, dùng thư viện Python nhẹ (hoặc backend xử lý) để trích xuất khuôn mặt thành \textbf{1 vector đặc trưng} (chuỗi số mã hóa).
    \item Server lưu chuỗi số này vào \textbf{Database}, đồng thời gửi vector xuống lưu vào \textbf{Flash/NVS của ESP32-S3-CAM} qua Wi-Fi (\textbf{MQTT} hoặc \textbf{HTTP}).
    \item Khi có người đứng trước cửa, \textbf{cảm biến hồng ngoại} xác nhận đó là vật thể 3D thật (không phải ảnh phẳng), ESP32-S3-CAM mới \textbf{tự chụp hình}, \textbf{tự trích xuất vector} của người đang đứng đó, rồi \textbf{so sánh trực tiếp} với các vector mẫu đã đồng bộ. Nếu khớp $\to$ chip \textbf{tự xoay Servo} để mở cửa.
  \end{enumerate}
\end{frame}

\begin{frame}{Chống giả mạo bằng cảm biến hồng ngoại}
  \textbf{Vấn đề:} nếu chỉ dựa vào camera, hệ thống có thể bị qua mặt bằng \textbf{ảnh in} hoặc \textbf{màn hình điện thoại} -- AI vẫn nhận đúng khuôn mặt và mở cửa.

  \vspace{0.6em}
  \textbf{Giải pháp:} thêm một lớp xác thực vật lý bằng \textbf{cảm biến hồng ngoại} (E18-D80NK hoặc TCRT5000) gắn ở vị trí trước cửa.

  \vspace{0.6em}
  \textbf{Cơ chế hoạt động:}
  \begin{itemize}
    \item Cảm biến phát tia hồng ngoại và đo lượng tín hiệu phản xạ về.
    \item Người thật là vật thể 3D có chiều sâu $\to$ tín hiệu phản xạ ổn định ở khoảng cách hợp lệ.
    \item Ảnh in / màn hình phẳng $\to$ đặc tính phản xạ khác, dễ rơi ngoài ngưỡng.
    \item Chỉ khi cảm biến xác nhận có vật thể thật, ESP32 mới kích hoạt camera và chạy AI.
  \end{itemize}

  \vspace{0.4em}
  \textbf{Lợi ích:} thêm 1 lớp chống giả mạo độc lập với AI, đồng thời \textbf{tiết kiệm năng lượng} -- camera + mô hình AI chỉ chạy khi thực sự có người tới gần.
\end{frame}

\begin{frame}{Ưu điểm}
  \begin{itemize}
    \item \textbf{Đáp ứng hoàn hảo tiêu chí AI + IoT:} đăng ký và quản lý tập trung qua mạng (IoT), nhưng xử lý nhận diện độc lập tại thiết bị (Edge AI).
    \vspace{0.4em}
    \item \textbf{Hoạt động Offline:} nếu công ty mất mạng Internet, thiết bị vẫn nhận diện và mở cửa bình thường cho 20 người đã được đồng bộ dữ liệu trước đó.
    \vspace{0.4em}
    \item \textbf{Bảo mật tuyệt đối:} hình ảnh gốc của nhân viên \textbf{chỉ nằm trên Server}; trên ESP32 chỉ lưu các \textbf{chuỗi số mã hóa}, không thể tái tạo lại ảnh khuôn mặt.
  \end{itemize}
\end{frame}

\begin{frame}{Các khó khăn có thể gặp phải}
  \begin{itemize}
    \item \textbf{Hiệu năng ESP32-S3 hạn chế:} chạy đồng thời 2 mô hình AI (Detect + Recognize) đòi hỏi tinh chỉnh kỹ về RAM/PSRAM và độ phân giải ảnh để đạt được phản hồi $<1.5$\,s.
    \vspace{0.3em}
    \item \textbf{Chất lượng ảnh ESP32-CAM thấp:} cảm biến nhạy với ánh sáng yếu, góc nghiêng, đeo kính/khẩu trang $\to$ ảnh hưởng tới độ chính xác nhận diện.
    \vspace{0.3em}
    \item \textbf{Cơ cấu chấp hành (mô hình bìa):} servo nhỏ, độ bền cơ học thấp $\to$ chỉ phù hợp demo, cần thay khóa điện từ thật khi triển khai.
  \end{itemize}
\end{frame}
