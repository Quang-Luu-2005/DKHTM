\begin{frame}{Giới thiệu đề tài}
  \footnotesize
  \begin{block}{Sản phẩm đề xuất}
    \textbf{Cửa an ninh thông minh} -- xác thực danh tính người dùng bằng khuôn mặt 2D, tự động mở/khóa cửa thông qua hệ thống thiết bị thông minh.
  \end{block}

  \vspace{0.25em}
  \textbf{Ý tưởng cốt lõi:}
  \begin{itemize}\setlength{\itemsep}{0.1em}
    \item Camera ESP32-CAM gắn tại cửa thu ảnh khuôn mặt người đứng trước
    \item Server chạy YOLOv11-Face detect mặt $\to$ ArcFace trích xuất embedding 512-d
    \item So khớp embedding với database đã đăng ký bằng cosine similarity
    \item Liveness detection (nháy mắt) chống giả mạo bằng ảnh in/video
    \item Nếu khớp + mặt thật $\to$ mở cửa; ngược lại $\to$ từ chối
  \end{itemize}

  \vspace{0.25em}
  \begin{alertblock}{Tại sao dùng AI face embedding?}
    So sánh pixel trực tiếp nhạy với ánh sáng, góc chụp và biểu cảm.
    ArcFace mã hóa khuôn mặt thành vector 512 chiều ổn định hơn và tập trung vào danh tính.
  \end{alertblock}
\end{frame}

\section{Related Work và lựa chọn công nghệ}
%============================

\begin{frame}{Lựa chọn công nghệ nhận dạng khuôn mặt}
  \footnotesize
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Các hướng đã khảo sát}
      \begin{itemize}\setlength{\itemsep}{0.25em}
        \item \textbf{dlib}: nhẹ, chạy CPU tốt nhưng độ chính xác thấp hơn các mô hình deep learning hiện đại.
        \item \textbf{FaceNet}: nền tảng tốt với triplet loss, nhưng training/tuning phức tạp hơn.
        \item \textbf{DeepFace}: wrapper tiện dùng nhiều model, nhưng nặng và khó tùy biến pipeline thiết bị.
      \end{itemize}

      \vspace{0.35em}
      \textbf{YOLOv11-Face -- detect/crop}
      \begin{itemize}\setlength{\itemsep}{0.25em}
        \item Tìm khuôn mặt nhanh trong frame ESP32-CAM.
        \item Crop đúng vùng mặt để đưa về input chuẩn cho ArcFace.
      \end{itemize}

    \column{0.48\textwidth}
      \textbf{ArcFace -- lõi nhận dạng}
      \begin{itemize}\setlength{\itemsep}{0.25em}
        \item Mô hình pretrained sẵn, chỉ cần inference, không cần train trên phần cứng.
        \item Trích xuất embedding 512 chiều ổn định với ánh sáng, góc chụp và biểu cảm.
        \item Accuracy khoảng 99.8\% trên benchmark LFW.
        \item Thư viện \texttt{insightface} hỗ trợ Python tốt.
      \end{itemize}

      \vspace{0.35em}
      \textbf{Liveness Detection -- chống giả mạo}
      \begin{itemize}\setlength{\itemsep}{0.25em}
        \item Dùng thử thách nháy mắt dựa trên Eye Aspect Ratio (EAR).
        \item Không cần thêm camera depth/IR, phù hợp prototype ESP32-CAM.
      \end{itemize}
  \end{columns}

  \vspace{0.8em}
  \textbf{Tóm lại:}
  ESP32-CAM chỉ thu ảnh và điều khiển relay; nhận dạng và chống giả mạo chạy trên laptop/server.
\end{frame}

\section{Pipeline AI nhận dạng khuôn mặt}
%============================

\begin{frame}{Pipeline xử lý AI -- Tổng quan}
  \textbf{Quy trình 4 bước}

  \vspace{0.4em}
  \begin{center}
    \footnotesize
    \texttt{Ảnh 2D}
    $\xrightarrow{\text{\textbf{1.} YOLOv11}}$ \texttt{Crop mặt}
    $\xrightarrow{\text{\textbf{2.} ArcFace}}$ \texttt{Embedding 512-d}
    $\xrightarrow{\text{\textbf{3.} Liveness}}$ \texttt{Mặt thật?}
    $\xrightarrow{\text{\textbf{4.} Match}}$ \texttt{PASS/DENY}
  \end{center}

  \vspace{0.8em}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Input}
      \begin{itemize}
        \item Ảnh JPEG từ ESP32-CAM
        \item Metadata: device\_id, timestamp
        \item Ảnh có thể mờ, thiếu sáng hoặc góc nghiêng nhẹ
      \end{itemize}

    \column{0.48\textwidth}
      \textbf{Output}
      \begin{itemize}
        \item Embedding khuôn mặt 512 chiều
        \item Điểm similarity với database
        \item Kết quả liveness
        \item Lệnh relay: mở hoặc từ chối
      \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Pipeline xử lý AI -- YOLOv11-Face}
  \textbf{Bước 1: Phát hiện và crop khuôn mặt}

  \vspace{0.3em}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \begin{itemize}
        \item Phát hiện bounding box khuôn mặt trong frame.
        \item Crop với hệ số mở rộng $s = 1.2$:
          \[
            \text{Size} = \max(w, h) \times 1.2
          \]
        \item Xử lý được nhiều mặt trong một frame.
        \item Nhanh, phù hợp real-time trên server.
      \end{itemize}

    \column{0.42\textwidth}
      \textbf{Đầu ra cho bước sau}
      \begin{itemize}
        \item Ảnh mặt đã crop
        \item Resize về kích thước input của ArcFace
        \item Loại bớt nền và người đứng xung quanh
      \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Pipeline xử lý AI -- ArcFace}
  \footnotesize
  \textbf{Bước 2: Trích xuất đặc trưng nhận dạng}

  \vspace{0.2em}
  ArcFace mã hóa khuôn mặt thành vector embedding 512 chiều:
  \begin{itemize}\setlength{\itemsep}{0.15em}
    \item Bất biến tương đối với ánh sáng, góc chụp và biểu cảm.
    \item Chỉ phụ thuộc vào danh tính (identity).
    \item Pretrained trên MS1MV2, dùng inference, không cần train lại trên ESP32-CAM.
  \end{itemize}

  \vspace{0.25em}
  ArcFace loss:
  \[
    L_i = -\log
    \frac{e^{s\cos(\theta_{y_i}+m)}}
    {e^{s\cos(\theta_{y_i}+m)} + \sum_{j \ne y_i} e^{s\cos(\theta_j)}}
  \]
  \[
    m = 0.5,\quad s = 64
  \]

  \vspace{0.1em}
  \textbf{Xác thực:}
  \[
    \mathrm{sim}(e_{\text{input}}, e_{\text{DB}})
    =
    \frac{e_{\text{input}} \cdot e_{\text{DB}}}
    {\lVert e_{\text{input}} \rVert \lVert e_{\text{DB}} \rVert}
  \]
  Nếu $\mathrm{sim} \geq \theta$ (ví dụ $0.5$) $\to$ \texttt{PASS}; ngược lại $\to$ \texttt{DENY}.
\end{frame}

\section{Chống giả mạo (Liveness Detection)}
%============================

\begin{frame}{Liveness Detection -- Nháy mắt}
  \footnotesize
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Mục đích}
      \begin{itemize}\setlength{\itemsep}{0.2em}
        \item Chống giả mạo bằng ảnh in, video phát lại hoặc màn hình điện thoại.
        \item Kiểm tra người trước cửa là mặt thật trước khi so khớp hoặc mở relay.
      \end{itemize}

      \vspace{0.3em}
      \textbf{Eye Aspect Ratio (EAR)}
      \[
        \mathrm{EAR} =
        \frac{\lVert p_2-p_6\rVert + \lVert p_3-p_5\rVert}
        {2\lVert p_1-p_4\rVert}
      \]
      \begin{itemize}\setlength{\itemsep}{0.15em}
        \item EAR bình thường $\approx 0.25$--$0.30$
        \item EAR $< 0.2$ $\to$ mắt nhắm / đang nháy
      \end{itemize}

    \column{0.48\textwidth}
      \textbf{Flow kiểm tra}
      \begin{enumerate}\setlength{\itemsep}{0.2em}
        \item Yêu cầu user nháy mắt qua LED/LCD.
        \item Theo dõi EAR trong 2--3 giây.
        \item Nếu phát hiện nháy $\to$ mặt thật $\to$ tiếp tục xác thực.
        \item Nếu không nháy trong 3s $\to$ nghi giả mạo $\to$ \texttt{DENY}.
      \end{enumerate}

      \vspace{0.3em}
      \textbf{Tại sao hiệu quả?}
      \begin{itemize}\setlength{\itemsep}{0.2em}
        \item Ảnh in không nháy mắt được.
        \item Có thể thêm random challenge: nháy mắt, quay trái/phải.
        \item Không cần thêm depth sensor hoặc camera IR.
      \end{itemize}
  \end{columns}
\end{frame}

\section{Rủi ro, thách thức và hướng giải quyết}
%============================

\begin{frame}{Rủi ro / thách thức và hướng giải quyết}
  \scriptsize
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Chất lượng ảnh ESP32-CAM}
      \begin{itemize}\setlength{\itemsep}{0.15em}
        \item Ảnh ban đêm yếu sáng, mờ, nhiễu hoặc ngược sáng.
        \item Khuôn mặt nhỏ, lệch góc, bị che bởi khẩu trang/nón.
        \item Ảnh 640$\times$480 vẫn dùng được vì ArcFace nhận ảnh crop/resize 112$\times$112.
        \item \textbf{Xử lý:} thêm LED, kiểm tra chất lượng ảnh, chụp lại nhiều frame và crop đúng mặt.
      \end{itemize}

      \vspace{0.25em}
      \textbf{Độ trễ hệ thống}
      \begin{itemize}\setlength{\itemsep}{0.15em}
        \item Latency mạng khi ESP32-CAM gửi ảnh JPEG lên server.
        \item Inference có thể vượt mục tiêu phản hồi nhanh.
        \item \textbf{Xử lý:} giảm kích thước ảnh, timeout rõ ràng, cache model và log thời gian từng bước.
      \end{itemize}

    \column{0.48\textwidth}
      \textbf{Sai nhận dạng}
      \begin{itemize}\setlength{\itemsep}{0.15em}
        \item False positive: mở cửa cho người không hợp lệ.
        \item False negative: từ chối người hợp lệ khi ảnh xấu.
        \item \textbf{Xử lý:} tinh chỉnh ngưỡng $\theta$, chụp lại khi confidence thấp, lưu ảnh sự kiện.
      \end{itemize}

      \vspace{0.25em}
      \textbf{Spoofing và bảo mật}
      \begin{itemize}\setlength{\itemsep}{0.15em}
        \item Ảnh in hoặc ảnh trên điện thoại có thể đánh lừa nhận dạng 2D.
        \item \textbf{Xử lý:} liveness bằng nháy mắt; có thể thêm random challenge quay trái/phải.
        \item Dữ liệu khuôn mặt là dữ liệu sinh trắc nhạy cảm.
        \item Xác thực device\_id/token, giới hạn quyền dashboard và mã hóa dữ liệu.
      \end{itemize}
  \end{columns}
\end{frame}

\section{Dataset và cách đánh giá}
%============================

\begin{frame}{Dataset kiểm thử}
  \footnotesize
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Dataset công khai}
      \begin{itemize}\setlength{\itemsep}{0.25em}
        \item \textbf{LFW} \cite{lfw}: benchmark chuẩn cho face verification.
        \item ArcFace pretrained đạt khoảng 99.8\% trên LFW.
        \item Dùng để tham khảo độ chính xác nhận dạng trước khi test thực tế.
      \end{itemize}

      \vspace{0.35em}
      \textbf{Dataset tự chụp}
      \begin{itemize}\setlength{\itemsep}{0.25em}
        \item Chụp bằng ESP32-CAM trong đúng bối cảnh cửa ra/vào.
        \item Mỗi người: 5--10 ảnh đăng ký, gồm chính diện, nghiêng nhẹ, sáng/tối.
        \item Tập test gồm ảnh mới của người đã đăng ký và ảnh người lạ.
      \end{itemize}

    \column{0.48\textwidth}
      \textbf{Chia tập thử nghiệm}
      \begin{itemize}\setlength{\itemsep}{0.25em}
        \item \textbf{Enroll set}: ảnh đăng ký khuôn mặt hợp lệ.
        \item \textbf{Test same-user}: ảnh cùng người nhưng khác góc/ánh sáng.
        \item \textbf{Test unknown}: ảnh người không có trong database.
        \item \textbf{Test spoof}: ảnh in hoặc ảnh trên điện thoại để test liveness.
      \end{itemize}

      \vspace{0.35em}
      \textbf{Kịch bản kiểm thử}
      \begin{itemize}\setlength{\itemsep}{0.25em}
        \item Kiểm thử offline trên laptop/server trước.
        \item Sau đó kiểm thử end-to-end: ESP32-CAM $\to$ server $\to$ relay/log.
        \item Ghi lại ảnh, kết quả, confidence và thời gian xử lý.
      \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Chỉ số đánh giá}
  \footnotesize
  \begin{columns}[T]
    \column{0.5\textwidth}
      \textbf{Xác thực danh tính}
      \begin{itemize}\setlength{\itemsep}{0.2em}
        \item \textbf{Accuracy}: tỷ lệ quyết định đúng trên tập tự chụp.
        \item \textbf{FAR}: tỷ lệ người lạ bị nhận nhầm là hợp lệ.
        \item \textbf{FRR}: tỷ lệ người hợp lệ bị từ chối.
        \item Đánh giá ở nhiều ngưỡng similarity $\theta$ khác nhau.
        \item Ma trận nhầm lẫn: \texttt{PASS}/\texttt{DENY} so với nhãn thật.
      \end{itemize}

      \vspace{0.25em}
      \[
        \text{FAR} = \frac{\text{False Accept}}{\text{Unknown Attempts}},
        \quad
        \text{FRR} = \frac{\text{False Reject}}{\text{Valid Attempts}}
      \]

    \column{0.46\textwidth}
      \textbf{Chất lượng pipeline}
      \begin{itemize}\setlength{\itemsep}{0.2em}
        \item Tỷ lệ phát hiện được mặt trong frame.
        \item Liveness detection rate: \% phát hiện đúng mặt thật vs giả.
        \item Similarity score trung bình theo từng điều kiện ánh sáng/góc nhìn.
      \end{itemize}

      \vspace{0.25em}
      \textbf{Hiệu năng hệ thống}
      \begin{itemize}\setlength{\itemsep}{0.2em}
        \item Tổng latency: chụp ảnh + gửi mạng + inference + trả lệnh.
        \item Mục tiêu phản hồi: dưới 3 giây trong điều kiện mạng ổn định.
        \item Tỷ lệ lỗi kết nối và số lần timeout.
      \end{itemize}
  \end{columns}
\end{frame}
